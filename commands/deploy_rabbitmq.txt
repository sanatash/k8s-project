helm install rabbitmq bitnami/rabbitmq --set image.repository=rabbitmq,image.tag=3-management --set auth.username=admin,auth.password=secretpassword --set metrics.enabled=true
helm uninstall rabbitmq
kubectl get pvc
# delete appropriate pvc and it will remove appropriate pv. uninstall doesn't delete rabbitmq persistent volume
# instead of uninstall and install again, you can do 'upgrade -i'
helm upgrade -i rabbitmq bitnami/rabbitmq --set image.repository=rabbitmq,image.tag=3-management --set auth.username=admin,auth.password=secretpassword --set metrics.enabled=true
